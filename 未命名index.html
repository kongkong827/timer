    <div class="timer-container">

        <div id="timer">15:00</div>

        <div class="buttons">

            <button id="resetBtn" onclick="resetTimer()">重置</button>

            <button id="startBtn" onclick="toggleTimer()">开始</button>

        </div>

    </div>


    <script>

        let timer;

        let timeLeft = 900; // 15分钟（单位：秒）

        let isRunning = false;


        function updateDisplay() {

            const minutes = Math.floor(timeLeft / 60).toString().padStart(2, '0');

            const seconds = (timeLeft % 60).toString().padStart(2, '0');

            document.getElementById('timer').textContent = `${minutes}:${seconds}`;

        }


        function toggleTimer() {

            if (isRunning) {

                clearInterval(timer);

                document.getElementById('startBtn').textContent = '继续';

            } else {

                document.getElementById('startBtn').textContent = '暂停';

                timer = setInterval(() => {

                    timeLeft--;

                    updateDisplay();

                    if (timeLeft <= 0) {

                        clearInterval(timer);

                        alert('时间到！');

                        resetTimer();

                    }

                }, 1000);

            }

            isRunning = !isRunning;

        }


        function resetTimer() {

            clearInterval(timer);

            timeLeft = 900;

            updateDisplay();

            isRunning = false;

            document.getElementById('startBtn').textContent = '开始';

        }


        // 初始化显示

        updateDisplay();

    </script>
